        let currentQuestionIndex = 0;
        let score = 0;

        function showAboutScreen() {
            switchScreen('screen-initial', 'about-screen');
        }

        function backToHome() {
            switchScreen('about-screen', 'screen-initial');
        }

        function startQuiz() {
            score = 0;
            currentQuestionIndex = 0;
            switchScreen('screen-initial', 'screen-quiz', () => {
                loadQuestion();
            });
        }

        function loadQuestion() {
            const questionData = questions[currentQuestionIndex];
            
            // Header e Progresso
            document.getElementById('question-counter').innerText = `Questão ${currentQuestionIndex + 1} de ${questions.length}`;
            const progress = ((currentQuestionIndex) / questions.length) * 100;
            document.getElementById('progress-bar').style.width = `${progress}%`;

            // Enunciado
            const questionEl = document.getElementById('question-text');
            questionEl.innerHTML = questionData.question.replace(/\n/g, '<br>');

            // Alternativas
            const alternativesContainer = document.getElementById('alternatives-container');
            alternativesContainer.innerHTML = '';
            
            questionData.alternatives.forEach((altText, index) => {
                const btn = document.createElement('button');
                btn.className = 'comic-btn alternative-btn';
                btn.innerText = altText;
                btn.setAttribute('aria-label', `Alternativa: ${altText}`);
                btn.setAttribute('tabindex', '0');
                btn.onclick = () => checkAnswer(index, btn);
                alternativesContainer.appendChild(btn);
            });

            // Ocultar feedback e botão próximo
            document.getElementById('feedback-container').classList.add('hidden');
            document.getElementById('next-btn-container').classList.add('hidden');
        }

                function checkAnswer(selectedIndex, selectedBtn) {
            const questionData = questions[currentQuestionIndex];
            const isCorrect = (selectedIndex === questionData.correct);
            const feedbackData = questionData.feedback[selectedIndex];

            if (isCorrect) score++;

            // Desabilitar todas as opções
            const buttons = document.querySelectorAll('.alternative-btn');
            buttons.forEach((btn, index) => {
                btn.classList.add('answer-disabled');
                
                if (index === selectedIndex) {
                    // Animacao de clique
                    btn.style.transform = 'scale(0.98)';
                    setTimeout(() => {
                        btn.style.transform = 'scale(1)';
                    }, 200);

                    if (isCorrect) {
                        btn.classList.add('answer-correct');
                        btn.innerHTML += '<span style="position: absolute; right: 16px; font-weight: bold; font-family: sans-serif;">✔</span>';
                    } else {
                        btn.classList.add('answer-wrong');
                        btn.innerHTML += '<span style="position: absolute; right: 16px; font-weight: bold; font-family: sans-serif;">✖</span>';
                    }
                }
            });

            // Somente após mudança e animacao, exibe feedback
            setTimeout(() => {
                showFeedback(feedbackData);
            }, 250);
        }

        function showFeedback(feedbackData) {
            const feedbackContainer = document.getElementById('feedback-container');
            const feedbackIcon = document.getElementById('feedback-icon');
            const feedbackTitle = document.getElementById('feedback-title');
            const feedbackText = document.getElementById('feedback-text');
            const nextBtnContainer = document.getElementById('next-btn-container');

            feedbackContainer.classList.remove('hidden', 'correct', 'incorrect');
            
            if (feedbackData.correct) {
                feedbackContainer.classList.add('correct');
                feedbackIcon.innerText = '✔';
            } else {
                feedbackContainer.classList.add('incorrect');
                feedbackIcon.innerText = '✖';
            }

            feedbackTitle.innerText = feedbackData.title;
            feedbackText.innerText = feedbackData.text;

            // Animação de entrada
            feedbackContainer.style.animation = 'none';
            feedbackContainer.offsetHeight; 
            feedbackContainer.style.animation = null;

            setTimeout(() => {
                nextBtnContainer.classList.remove('hidden');
                
                // Progresso atualizado para indicar que a questão foi respondida
                const progress = ((currentQuestionIndex + 1) / questions.length) * 100;
                document.getElementById('progress-bar').style.width = `${progress}%`;
                
                nextBtnContainer.querySelector('button').focus();
            }, 300);
        }

        function nextQuestion() {
            currentQuestionIndex++;
            
            if (currentQuestionIndex < questions.length) {
                const quizScreen = document.getElementById('screen-quiz');
                quizScreen.style.opacity = '0';
                
                setTimeout(() => {
                    loadQuestion();
                    quizScreen.style.opacity = '1';
                }, 300);
            } else {
                finishQuiz();
            }
        }

        function finishQuiz() {
            const percentage = Math.round((score / questions.length) * 100);
            
            document.getElementById('final-score-text').innerHTML = `Você acertou<br>${score} de ${questions.length} questões!`;
            document.getElementById('final-percentage').innerText = `${percentage}%`;
            
            let message = "";
            if (percentage <= 30) {
                message = "Continue estudando! Cada tentativa é uma oportunidade para aprender mais sobre os pronomes relativos.";
            } else if (percentage <= 60) {
                message = "Bom trabalho! Você já domina parte do conteúdo. Revise os pontos em que teve dificuldade e tente novamente.";
            } else if (percentage < 100) {
                message = "Muito bem! Você demonstrou um bom conhecimento sobre os pronomes relativos. Continue praticando para alcançar a pontuação máxima.";
            } else {
                message = "Parabéns! Você acertou todas as questões e demonstrou excelente domínio dos pronomes relativos.";
            }
            
            document.getElementById('final-message').innerText = message;
            
            switchScreen('screen-quiz', 'screen-final');
        }

        function restartQuiz() {
            switchScreen('screen-final', 'screen-initial');
        }

        function switchScreen(hideId, showId, callback) {
            const hideEl = document.getElementById(hideId);
            const showEl = document.getElementById(showId);
            
            hideEl.classList.remove('active');
            
            setTimeout(() => {
                if (callback) callback();
                showEl.classList.add('active');
            }, 300);
        }
    </script>
